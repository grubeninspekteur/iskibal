== Eligibility

.ELIG_001: Age check
[source,iskara,.rule]
----
when
    customer.age >= 18
then
    eligible := "yes"
else
    eligible := "no"
end
----

== Coupon Codes

// Template rule using an inline data table
[source,iskara,.rule]
----
template rule COUPON "Coupon code discount"
data table {
    | code      | amount |
    | --------- | ------ |
    | "PROMO10" | 10     |
    | "PROMO25" | 25     |
    | "PROMO50" | 50     |
}
when
    order.couponCode = code
then
    discountPercent := amount
end
----
